- git branch	//list all available branches
- git checkout [branch name]	//move to yourbranch

- git status	//status of your local working environment
- git status -s	//a more summarised status

- git add /path/to/file
- git add /path/to/directory/*	//add all untracked files (The ones with the ?)

- git commit -am "commit message"	//very obvious
- git push -u origin [your_current_working_branch]	//There are different ways of pushing, this is just how I do it
